def process_and_labeling(img):
    X=[]
    y=[]
    
    for image in img:
        X.append(cv2.resize(cv2.imread(image, cv2.IMREAD_COLOR), (nrows,ncloumns),interpolation=cv2.INTER_CUBIC))
        
        if 'other' in image:
            y.append(0)
        elif 'music' in image:
            y.append(1)
        
    return X,y
